CREATE OR REPLACE FUNCTION GET_MAX_PK(TABLE_NAME VARCHAR) RETURN NUMBER IS

        V_MAX_PK NUMBER ;
BEGIN 
 
	EXECUTE IMMEDIATE 'SELECT NVL(MAX('||SUBSTR(TABLE_NAME, 1, LENGTH(TABLE_NAME) - 1)||'_ID),0) FROM '|| TABLE_NAME INTO V_MAX_PK  ;
	RETURN V_MAX_PK ;
	EXCEPTION WHEN NO_DATA_FOUND THEN
		RETURN NULL;
END;
